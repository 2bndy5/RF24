examples_linux/multiceiver_demo.py

This is a simple example of using the RF24 class on a Raspberry Pi for using 1 nRF24L01 to receive data from up to 6 other transceivers. This technique is called "multiceiver" in the datasheet.

Remember to install the Python wrapper, then navigate to the "RF24/examples_linux" folder and open a python3 REPL.
To run this example, enter

1 from multiceiver_demo import *

and follow the prompts.

Note
this example requires python v3.7 or newer because it measures transmission time with time.monotonic_ns().

23 # setup the addresses for all transmitting nRF24L01 nodes
24 addresses = [
25  b"\x78" * 5,
26  b"\xF1\xB6\xB5\xB4\xB3",
27  b"\xCD\xB6\xB5\xB4\xB3",
28  b"\xA3\xB6\xB5\xB4\xB3",
29  b"\x0F\xB6\xB5\xB4\xB3",
30  b"\x05\xB6\xB5\xB4\xB3"

32 # It is very helpful to think of an address as a path instead of as
33 # an identifying device destination
34 
35 # initialize the nRF24L01 on the spi bus
36 if not radio.begin():
37  raise RuntimeError("nRF24L01 hardware isn't responding")
38 
39 # set the Power Amplifier level to -12 dBm since this test example is
40 # usually run with nRF24L01 transceivers in close proximity of each other
41 radio.setPALevel(RF24_PA_LOW) # RF24_PA_MAX is default
42 
43 # To save time during transmission, we'll set the payload size to be only what
44 # we need.
45 # A byte and an int occupy 5 bytes in memory using len(struct.pack())
46 # "<bi" means a little endian unsigned byte and int
47 radio.payloadSize = len(struct.pack("<ii", 0, 0))
48 
49 # for debugging
50 radio.printDetails()

53 def base(timeout=10):
54  """Use the nRF24L01 as a base station for lisening to all nodes"""
55  # write the addresses to all pipes.
56  for pipe_n, addr in enumerate(addresses):
57  radio.openReadingPipe(pipe_n, addr)
58  radio.startListening() # put base station into RX mode
59  start_timer = time.monotonic() # start timer
60  while time.monotonic() - start_timer < timeout:
61  has_payload, pipe_number = radio.available_pipe()
62  if has_payload:
63  # unpack payload
64  nodeID, payloadID = struct.unpack(
65  "<ii",
66  radio.read(radio.payloadSize)
67  )
68  # show the pipe number that received the payload
69  print(
70  "Received {} bytes on pipe {} from node {}. PayloadID: "
71  "{}".format(
72  radio.payloadSize,
73  pipe_number,
74  nodeID,
75  payloadID
76  )
77  )
78  start_timer = time.monotonic() # reset timer with every payload
79  radio.stopListening()

82 def node(node_number, count=10):
83  """start transmitting to the base station.
84 
85  :param int node_number: the node's identifying index (from the
86  the `addresses` list). This is a required parameter
87  :param int count: the number of times that the node will transmit
88  to the base station.
89  """
90  radio.stopListening()
91  # set the TX address to the address of the base station.
92  radio.openWritingPipe(addresses[node_number])
93  counter = 0
94  # use the node_number to identify where the payload came from
95  while counter < count:
96  counter += 1
97  # payloads will include the node_number and a payload ID character
98  payload = struct.pack("<ii", node_number, counter)
99  start_timer = time.monotonic_ns()
100  report = radio.write(payload)
101  end_timer = time.monotonic_ns()
102  # show something to see it isn't frozen
103  if report:
104  print(
105  "Transmission of payloadID {} as node {} successfull! "
106  "Transmission time: {} us".format(
107  counter,
108  node_number,
109  (end_timer - start_timer) / 1000
110  )
111  )
112  else:
113  print("Transmission failed or timed out")
114  time.sleep(0.5) # slow down the test for readability
