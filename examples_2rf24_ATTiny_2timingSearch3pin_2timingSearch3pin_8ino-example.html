123 /****************************************************************************/
124 void csn(bool mode) {
125  if (ce_pin != csn_pin) {
126  digitalWrite(csn_pin, mode);
127  } else {
128  if (mode == HIGH) {
129  PORTB |= (1 << PINB2); // SCK->CSN HIGH
130  delayMicroseconds(csnHighSettle); // allow csn to settle
131  } else {
132  PORTB &= ~(1 << PINB2); // SCK->CSN LOW
133  delayMicroseconds(csnLowSettle); // allow csn to settle
134  }
135  }
136 }

139 uint8_t read_register(uint8_t reg)
140 {
141  csn(LOW);
142  SPI.transfer( R_REGISTER | ( REGISTER_MASK & reg ) );
143  uint8_t result = SPI.transfer(0xff);

173 void setup(void) {
174  uint8_t status;
175 
176  // start serial port and SPI
177  Serial.begin(9600);
178  SPI.begin();
179 
180  // configure ce and scn as output when used
181  ce_pin = CE_PIN;
182  csn_pin = CSN_PIN;
183 
184  setCsnHighSettle(MAX_HIGH);
185  setCsnLowSettle(MAX_LOW);
186  // csn is used in SPI transfers. Set to LOW at start and HIGH after transfer. Set to HIGH to reflect no transfer active
187  // SPI command are accepted in Power Down state.
188  // ce represent PRX (LOW) or PTX (HIGH) mode apart from register settings. Start in PRX mode.
189  ce(LOW);
190  csn(HIGH);
191 
192  // nRF24L01 goes from to Power Down state 100ms after Power on Reset ( Vdd > 1.9V) or when PWR_UP is 0 in config register
193  // Goto Power Down state (Powerup or force) and set in transmit mode
194  write_register2(NRF_CONFIG, read_register(NRF_CONFIG) & ~_BV(PWR_UP) & ~_BV(PRIM_RX));
195  delay(100);
196 
197  // Goto Standby-I
198  // Technically we require 4.5ms Tpd2stby+ 14us as a worst case. We'll just call it 5ms for good measure.
199  // WARNING: Delay is based on P-variant whereby non-P *may* require different timing.
200  write_register2(NRF_CONFIG, read_register(NRF_CONFIG) | _BV(PWR_UP));
201  delay(5) ;
202 
203  // Goto Standby-II
204  ce(HIGH);
205  Serial.print("Scanning for optimal setting time for scn");
206 }

209 void loop(void) {
210  uint8_t status;
211  uint8_t i;
212  uint8_t j;
213  uint8_t k;
214  bool success = true;
215  uint8_t csnHigh = MAX_HIGH;
216  uint8_t csnLow = MAX_LOW;
217  uint8_t bottom_success;
218  bool bottom_found;
219  uint8_t value[] = {5, 10};
220  uint8_t limit[] = {MAX_HIGH, MAX_LOW};
221  uint8_t advice[] = {MAX_HIGH, MAX_LOW};
222 
223  // check max values give correct behavior
224  for (k = 0; k < 2; k++) {
225  bottom_found = false;
226  bottom_success = 0;
227  while (bottom_success < 255) {
228  setCsnHighSettle(limit[0]);
229  setCsnLowSettle(limit[1]);
230  // check current values
231  i = 0;
232  while (i < 255 & success) {
233  for (j = 0; j < 2; j++) {
234  write_register2(EN_AA, value[j]);
235  status = read_register(EN_AA);
236  if (value[j] != status) {
237  success = false;
238  }
239  }
240  i++;
241  }
242  // process result of current values
243  if (!success) {
244  Serial.print("Settle NOK. csnHigh=");
245  Serial.print(limit[0], DEC);
246  Serial.print(" csnLow=");
247  Serial.println(limit[1], DEC);
248  limit[k]++;
249  bottom_found = true;
250  bottom_success = 0;
251  success = true;
252  } else {
253  Serial.print("Settle OK. csnHigh=");
254  Serial.print(limit[0], DEC);
255  Serial.print(" csnLow=");
256  Serial.println(limit[1], DEC);
257  if (!bottom_found) {
258  limit[k]--;
259  if (limit[k] == MINIMAL) {
260  bottom_found = true;
261  bottom_success = 0;
262  success = true;
263  }
264  } else {
265  bottom_success++;
266  }
267  }
268  }
269  Serial.print("Settle value found for ");
270  if (k == 0) {
271  Serial.print("csnHigh: ");
272  } else {
273  Serial.print("csnLow: ");
274  }
275  Serial.println(limit[k], DEC);
276  advice[k] = limit[k] + (limit[k] / 10);
277  limit[k] = 100;
278  }
279  Serial.print("Adviced Settle times are: csnHigh=");
280  Serial.print(advice[0], DEC);
281  Serial.print(" csnLow=");
282  Serial.println(advice[1], DEC);
283  while (true)
284  {
285  ;
286  }
287 }
